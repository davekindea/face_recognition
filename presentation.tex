\section{Introduction \& Project Goal}
% ══════════════════════════════════════════════════════════════════════════════

\begin{frame}{Introduction \& Project Goal}
  \begin{columns}
    \begin{column}{0.5\textwidth}
      \begin{block}{The Goal}
        Build an automated end-to-end system to recognize faces of specific individuals with high confidence.
      \end{block}
      \begin{block}{Challenges}
        \begin{itemize}
          \item Small datasets per person (9--13 images)
          \item Varied illumination, poses, and settings
          \item Different image sizes, formats, and channels
        \end{itemize}
      \end{block}
    \end{column}
    \begin{column}{0.5\textwidth}
      \begin{block}{Proposed Pipeline}
        \begin{enumerate}
          \item \textbf{Face Detection}: MediaPipe BlazeFace
          \item \textbf{Feature Extraction}: 128-D FaceNet Embeddings
          \item \textbf{Normalization}: L2 Norm
          \item \textbf{Classification}: Support Vector Machine (SVM)
          \item \textbf{Web Deployment}: Streamlit Interface
        \end{enumerate}
      \end{block}
    \end{column}
  \end{columns}
\end{frame}

\section{Image Preprocessing}
% ══════════════════════════════════════════════════════════════════════════════

\begin{frame}{Image Preprocessing with MediaPipe}
  \begin{columns}
    \begin{column}{0.5\textwidth}
      \begin{block}{The Preprocessing Pipeline}
        \begin{enumerate}
          \item Read raw BGR image using \textbf{OpenCV}
          \item Convert image to RGB format
          \item Detect face bounding boxes using \textbf{MediaPipe}
          \item Crop the primary face region
          \item Standardize size to \textbf{224 $\times$ 224 pixels}
          \item Save structured clean images
        \end{enumerate}
      \end{block}
    \end{column}
    \begin{column}{0.5\textwidth}
      \begin{block}{Key Detail}
        BlazeFace detector utilizes a short-range model optimized for fast mobile/web performance.
        \\[0.5em]
        Images where no face is detected are safely skipped to avoid contaminating the training set.
      \end{block}
    \end{column}
  \end{columns}
\end{frame}

\section{Feature Extraction --- FaceNet}
% ══════════════════════════════════════════════════════════════════════════════

\begin{frame}{Feature Extraction via FaceNet}
  \begin{columns}
    \begin{column}{0.55\textwidth}
      \begin{block}{What is FaceNet?}
        A deep convolutional network trained to directly map face images to a compact Euclidean space.
        \begin{itemize}
          \item Distances directly correspond to face similarity.
          \item Learns robust representation invariant to style/illumination.
        \end{itemize}
      \end{block}
      \begin{block}{Embeddings}
        \begin{itemize}
          \item Output representation: \textbf{128-D vector}
          \item Extracted using \texttt{DeepFace.represent}
        \end{itemize}
      \end{block}
    \end{column}
    \begin{column}{0.45\textwidth}
      \begin{block}{L2 Normalization (Critical)}
        All embeddings must be L2-normalized:
        \[
        \hat{x} = \frac{x}{\|x\|_2}
        \]
        This projects coordinates onto a unit hypersphere, matching how the FaceNet loss (triplet loss) was trained.
      \end{block}
    \end{column}
  \end{columns}
\end{frame}

\section{Classification --- Support Vector Machine}
% ══════════════════════════════════════════════════════════════════════════════

\begin{frame}{Classification with Linear SVM}
  \begin{columns}
    \begin{column}{0.5\textwidth}
      \begin{block}{Support Vector Machine (SVM)}
        Finds the optimal separating hyperplanes between the 10 identities in the 128-D normalized embedding space.
      \end{block}
      \begin{block}{Grid Search Optimization}
        \begin{itemize}
          \item Optimized using \texttt{GridSearchCV} with 5-Fold Cross-Validation.
          \item Tested kernels: \texttt{linear}, \texttt{rbf}.
          \item Tested penalty parameters $C$: \texttt{0.1, 1.0, 10.0}.
        \end{itemize}
      \end{block}
    \end{column}
    \begin{column}{0.5\textwidth}
      \begin{block}{Best Classifier Configuration}
        \begin{tabular}{ll}
          \toprule
          \textbf{Parameter} & \textbf{Selection}\\
          \midrule
          Kernel & \textbf{Linear}\\
          $C$ & \textbf{1.0}\\
          Class Weight & \textbf{Balanced}\\
          Cross-Val Acc. & \textbf{86.48\%}\\
          Training Acc. & \textbf{94.59\%}\\
          \bottomrule
        \end{tabular}
      \end{block}
    \end{column}
  \end{columns}
\end{frame}
